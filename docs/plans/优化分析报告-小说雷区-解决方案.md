```python
        """
        完整的审校流程：
        1. 原有事实/连贯性审校
        2. AI 味检测（新增）
        3. 风格一致性检查
        """
        # 第一步：原有审校
        base_issues = await self._base_review(content, constraint)

        # 第二步：AI 味检测
        ai_smell_issues = self._detect_ai_patterns(content)

        # 第三步：风格检查
        style_issues = await self._style_check(content, constraint)

        # 合并
        all_issues = base_issues + ai_smell_issues + style_issues

        # 分级处理
        critical = [i for i in all_issues if i.get("severity") == "critical"]
        major = [i for i in all_issues if i.get("severity") == "major"]

        if critical:
            return {
                "status": "needs_rewrite",
                "issues": all_issues,
                "rewrite_hints": self._build_rewrite_hints(critical + major)
            }

        if len(major) >= 3:
            return {
                "status": "needs_rewrite",
                "issues": all_issues,
                "rewrite_hints": self._build_rewrite_hints(major)
            }

        return {"status": "pass", "issues": all_issues}

    def _detect_ai_patterns(self, content: str) -> list[dict]:
        """
        扫描全文，返回所有命中的 AI 味问题列表。
        每个问题包含：id, name, severity, count, message, locations（上下文片段）
        """
        issues = []

        # --- Hard Ban 检测 ---
        hard_rules = self.style_rules["hard_ban_patterns"]["rules"]
        for rule in hard_rules:
            matches = list(re.finditer(rule["pattern"], content))
            if len(matches) >= rule["threshold"] + 1:  # +1 因为 threshold=0 表示一个都不允许
                # 收集匹配位置，取前后 30 字作为上下文
                locations = []
                for m in matches[:5]:  # 最多展示 5 处
                    start = max(0, m.start() - 30)
                    end = min(len(content), m.end() + 30)
                    locations.append(content[start:end].replace("\n", " "))

                issues.append({
                    "id": rule["id"],
                    "name": rule["name"],
                    "severity": "critical",
                    "count": len(matches),
                    "message": rule["message"],
                    "locations": locations
                })

        # --- Soft Ban 检测 ---
        soft_rules = self.style_rules["soft_ban_patterns"]["rules"]
        for rule in soft_rules:
            if "pattern" in rule:
                matches = list(re.finditer(rule["pattern"], content))
                if len(matches) > rule["threshold"]:
                    locations = []
                    for m in matches[:5]:
                        start = max(0, m.start() - 30)
                        end = min(len(content), m.end() + 30)
                        locations.append(content[start:end].replace("\n", " "))
                    issues.append({
                        "id": rule["id"],
                        "name": rule["name"],
                        "severity": "major",
                        "count": len(matches),
                        "message": rule["message"],
                        "locations": locations
                    })

        # --- 对话标签比例检测 ---
        dialogue_lines = re.findall(r"[「「"'].+?[」」"']", content)
        tag_count = len(re.findall(r"(?:说|道|问|答|喊|叫|吼|骂|嚷|呢喃|低语|耳语|嘀咕|嘟囔)[。，；：\n]", content))
        if dialogue_lines and tag_count / len(dialogue_lines) > 0.15:
            issues.append({
                "id": "DIALOGUE_TAG_RATE",
                "name": "对话标签比例",
                "severity": "major",
                "count": round(tag_count / len(dialogue_lines), 2),
                "message": f"对话标签比例 {tag_count/len(dialogue_lines):.0%}，超过 15%"
            })

        # --- 破折号密度检测 ---
        em_dash_count = content.count("——")
        char_count = len(content.replace("\n", "").replace(" ", ""))
        density = em_dash_count / (char_count / 1000) if char_count > 0 else 0
        if density > 1:
            issues.append({
                "id": "EM_DASH_OVERUSE",
                "name": "中文破折号密度",
                "severity": "major",
                "count": round(density, 1),
                "message": f"破折号密度 {density:.1f}/千字，超过阈值 1/千字"
            })

        # --- 成语密度检测（简化版）---
        chengyu_pattern = re.compile(r"[\u4e00-\u9fff]{4}(?=[，。；：！？\n])")
        chengyu_matches = [m for m in chengyu_pattern.finditer(content)]
        density = len(chengyu_matches) / (char_count / 500) if char_count > 0 else 0
        if density > 3:
            issues.append({
                "id": "CHENGYU_DENSITY",
                "name": "成语密度",
                "severity": "major",
                "count": round(density, 1),
                "message": f"成语密度 {density:.1f}/500字，超过阈值 3/500字"
            })

        # --- 句式多样性检测 ---
        sentences = re.split(r"[。！？\n]", content)
        lengths = [len(s) for s in sentences if 3 < len(s) < 200]
        if lengths:
            mean_len = sum(lengths) / len(lengths)
            variance = sum((l - mean_len) ** 2 for l in lengths) / len(lengths)
            std = variance ** 0.5
            if std < 5:
                issues.append({
                    "id": "SENTENCE_VARIANCE",
                    "name": "句式多样性",
                    "severity": "minor",
                    "count": round(std, 1),
                    "message": f"句子长度标准差 {std:.1f}，过于均匀"
                })

        return issues

    def _build_rewrite_hints(self, issues: list[dict]) -> str:
        """
        将检测到的问题转化为 Writer Agent 能理解的重写指令。
        """
        hints = ["## 重写指令：以下 AI 写作特征必须修复\n"]
        for issue in issues:
            hints.append(f"- [{issue['severity']}] {issue['name']}：{issue['message']}")
            if issue.get("locations"):
                hints.append(f"  违规片段：")
                for loc in issue["locations"][:2]:
                    hints.append(f"  「...{loc}...」")
        hints.append("\n请重写整个段落。不要修补——重新生成。")
        return "\n".join(hints)
```

```python
{hard}

### 严格限制的词汇（减少 90%）
{soft}
```

### 4.2 毛刺注入（V2 功能，Phase 7 StyleAnalyzer 扩展）

**文件路径**：`backend/app/services/style_analyzer.py`（新增方法）

```python
import random

class StyleAnalyzer:
    ROUGHNESS_STRATEGIES = [
        "break_paragraph_abruptly",       # 某段在中间戛然而止
        "incomplete_metaphor",            # 一个故意不完整的比喻
        "sentence_length_spike",          # 插入一句超短句（1-3字）或超长句（40+字）
        "raw_sensory_no_interpretation",  # 纯感官描写，不给解释
        "jarring_transition",             # 一处突兀的段落过渡（不做铺垫）
    ]

    async def inject_human_roughness(self, content: str, chapter_number: int) -> str:
        """
        随机注入 1-2 处"人工毛刺"，打破 AI 过于平滑的叙事。

        触发条件（满足任一即触发）：
        1. 审校检测到 EXCESSIVELY_SMOOTH 标记
        2. 每隔 5 章自动执行一次（保持节奏）
        """
        if chapter_number % 5 != 0:
            return content  # 非触发章节，跳过

        strategies = random.sample(self.ROUGHNESS_STRATEGIES, k=random.randint(1, 2))

        prompt = f"""以下是小说第 {chapter_number} 章的一部分正文。

请在 **1-2 处** 做微小的"不完美化"处理。不要改变情节，不要改变角色行为。

具体操作（只能选其中 {len(strategies)} 种）：
{chr(10).join(f"- {s}" for s in strategies)}

## 处理要求
- 每次修改只有 1-3 个句子受影响
- 读者不会明确注意到，但下意识会感到"这段更有人味"
- 不解释你做了什么

## 正文
{content}

处理后直接输出完整正文。"""

        return await self.llm.complete(prompt)
```
